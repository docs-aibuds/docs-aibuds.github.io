Best Practices

This guide provides best practices for using the AIBuds SDK in your iOS applications.

SDK Initialization​

Objective-C​

// Initialize the SDK in your app delegate
- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions {
    // Initialize AIBuds SDK
    [AIBudsSDK initializeWithAppKey:@"YOUR_APP_KEY" completion:^(BOOL success, NSError *error) {
        if (success) {
            NSLog(@"AIBuds SDK initialized successfully");
        } else {
            NSLog(@"Failed to initialize AIBuds SDK: %@", error.localizedDescription);
        }
    }];
    
    return YES;
}

Swift​

// Initialize the SDK in your app delegate
func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
    // Initialize AIBuds SDK
    AIBudsSDK.initialize(withAppKey: "YOUR_APP_KEY") { success, error in
        if success {
            print("AIBuds SDK initialized successfully")
        } else {
            print("Failed to initialize AIBuds SDK: \(error?.localizedDescription ?? "Unknown error")")
        }
    }
    
    return true
}

Device Management​

Connection Management​

  • Always check the connection state before performing operations on a device
  • Handle connection state changes appropriately
  • Disconnect from devices when they're no longer needed

Objective-C​

// Check connection state before operation
if (device.connectionState == AIBudsDeviceConnectionStateConnected) {
    // Perform operation
    [audioManager setNoiseCancellationLevel:AIBudsNoiseCancellationLevelHigh forDevice:device completion:^(BOOL success, NSError *error) {
        if (success) {
            NSLog(@"Noise cancellation level set successfully");
        } else {
            NSLog(@"Failed to set noise cancellation level: %@", error.localizedDescription);
        }
    }];
} else {
    NSLog(@"Device is not connected");
}

// Listen for connection state changes
[[NSNotificationCenter defaultCenter] addObserver:self
                                         selector:@selector(deviceConnectionStateChanged:)
                                             name:AIBudsDeviceConnectionStateDidChangeNotification
                                           object:nil];

- (void)deviceConnectionStateChanged:(NSNotification *)notification {
    AIBudsDevice *device = notification.userInfo[AIBudsDeviceKey];
    AIBudsDeviceConnectionState state = device.connectionState;
    
    switch (state) {
        case AIBudsDeviceConnectionStateConnected:
            NSLog(@"Device connected: %@", device.name);
            break;
        case AIBudsDeviceConnectionStateDisconnected:
            NSLog(@"Device disconnected: %@", device.name);
            break;
        case AIBudsDeviceConnectionStateConnecting:
            NSLog(@"Device connecting: %@", device.name);
            break;
        case AIBudsDeviceConnectionStateDisconnecting:
            NSLog(@"Device disconnecting: %@", device.name);
            break;
    }
}

Swift​

// Check connection state before operation
if device.connectionState == .connected {
    // Perform operation
    audioManager.setNoiseCancellationLevel(.high, for: device) { success, error in
        if success {
            print("Noise cancellation level set successfully")
        } else {
            print("Failed to set noise cancellation level: \(error?.localizedDescription ?? "Unknown error")")
        }
    }
} else {
    print("Device is not connected")
}

// Listen for connection state changes
NotificationCenter.default.addObserver(
    forName: .AIBudsDeviceConnectionStateDidChange,
    object: nil,
    queue: .main
) { notification in
    if let device = notification.userInfo?[AIBudsDeviceKey] as? AIBudsDevice {
        let state = device.connectionState
        
        switch state {
        case .connected:
            print("Device connected: \(device.name)")
        case .disconnected:
            print("Device disconnected: \(device.name)")
        case .connecting:
            print("Device connecting: \(device.name)")
        case .disconnecting:
            print("Device disconnecting: \(device.name)")
        }
    }
}

Audio Management​

Noise Cancellation​

  • Use the appropriate noise cancellation level based on the environment
  • Consider using smart noise cancellation for automatic adjustment

Objective-C​

// Set noise cancellation level based on environment
- (void)setNoiseCancellationLevelForEnvironment:(AIBudsEnvironmentType)environment {
    AIBudsNoiseCancellationLevel level;
    
    switch (environment) {
        case AIBudsEnvironmentTypeQuiet:
            level = AIBudsNoiseCancellationLevelLow;
            break;
        case AIBudsEnvironmentTypeModerate:
            level = AIBudsNoiseCancellationLevelMedium;
            break;
        case AIBudsEnvironmentTypeNoisy:
            level = AIBudsNoiseCancellationLevelHigh;
            break;
        default:
            level = AIBudsNoiseCancellationLevelMedium;
            break;
    }
    
    [audioManager setNoiseCancellationLevel:level forDevice:device completion:^(BOOL success, NSError *error) {
        if (success) {
            NSLog(@"Noise cancellation level set to %ld", (long)level);
        }
    }];
}

Swift​

// Set noise cancellation level based on environment
func setNoiseCancellationLevelForEnvironment(_ environment: AIBudsEnvironmentType) {
    var level: AIBudsNoiseCancellationLevel
    
    switch environment {
    case .quiet:
        level = .low
    case .moderate:
        level = .medium
    case .noisy:
        level = .high
    default:
        level = .medium
    }
    
    audioManager.setNoiseCancellationLevel(level, for: device) { success, error in
        if success {
            print("Noise cancellation level set to \(level)")
        }
    }
}

Equalizer Settings​

  • Provide users with preset equalizer options
  • Allow users to create and save custom equalizer settings
  • Remember user preferences across sessions

Objective-C​

// Save equalizer preset to user defaults
- (void)saveEqualizerPreset:(AIBudsEqualizerPreset)preset {
    [[NSUserDefaults standardUserDefaults] setInteger:preset forKey:@"EqualizerPreset"];
    [[NSUserDefaults standardUserDefaults] synchronize];
}

// Load equalizer preset from user defaults
- (AIBudsEqualizerPreset)loadEqualizerPreset {
    NSInteger preset = [[NSUserDefaults standardUserDefaults] integerForKey:@"EqualizerPreset"];
    return (AIBudsEqualizerPreset)preset;
}

// Apply saved equalizer preset
- (void)applySavedEqualizerPreset {
    AIBudsEqualizerPreset preset = [self loadEqualizerPreset];
    [audioManager setEqualizerPreset:preset forDevice:device completion:^(BOOL success, NSError *error) {
        if (success) {
            NSLog(@"Equalizer preset applied: %ld", (long)preset);
        }
    }];
}

Swift​

// Save equalizer preset to user defaults
func saveEqualizerPreset(_ preset: AIBudsEqualizerPreset) {
    UserDefaults.standard.set(preset.rawValue, forKey: "EqualizerPreset")
}

// Load equalizer preset from user defaults
func loadEqualizerPreset() -> AIBudsEqualizerPreset {
    let presetValue = UserDefaults.standard.integer(forKey: "EqualizerPreset")
    return AIBudsEqualizerPreset(rawValue: presetValue) ?? .normal
}

// Apply saved equalizer preset
func applySavedEqualizerPreset() {
    let preset = loadEqualizerPreset()
    audioManager.setEqualizerPreset(preset, for: device) { success, error in
        if success {
            print("Equalizer preset applied: \(preset)")
        }
    }
}

Battery Usage​

  • Disable unnecessary features when not in use
  • Use background modes appropriately
  • Monitor battery usage and provide user feedback

Objective-C​

// Optimize battery usage
- (void)optimizeBatteryUsage {
    // Disable AI features when not needed
    if (!userPrefersAIFeatures) {
        [aiManager setAIFeaturesEnabled:NO forDevice:device completion:^(BOOL success, NSError *error) {
            if (success) {
                NSLog(@"AI features disabled to save battery");
            }
        }];
    }
    
    // Disable voice assistant when not needed
    if (!userPrefersVoiceAssistant) {
        [vaManager setVoiceAssistantEnabled:NO];
        NSLog(@"Voice assistant disabled to save battery");
    }
}

Memory Usage​

  • Release resources when not needed
  • Use weak references to avoid retain cycles
  • Monitor memory usage and handle low memory warnings

Objective-C​

// Handle low memory warning
- (void)applicationDidReceiveMemoryWarning:(UIApplication *)application {
    // Release unnecessary resources
    [self cleanupResources];
}

- (void)cleanupResources {
    // Disconnect from devices if not actively using them
    NSArray<AIBudsDevice *> *connectedDevices = [deviceManager getConnectedDevices];
    for (AIBudsDevice *device in connectedDevices) {
        if (![self isDeviceActivelyUsed:device]) {
            [deviceManager disconnectFromDevice:device completion:^(BOOL success, NSError *error) {
                if (success) {
                    NSLog(@"Disconnected from device to save memory: %@", device.name);
                }
            }];
        }
    }
}

- (BOOL)isDeviceActivelyUsed:(AIBudsDevice *)device {
    // Check if device is actively being used
    // Return YES if actively used, NO otherwise
    return NO;
}

Swift​

// Handle low memory warning
func applicationDidReceiveMemoryWarning(_ application: UIApplication) {
    // Release unnecessary resources
    cleanupResources()
}

func cleanupResources() {
    // Disconnect from devices if not actively using them

                }
            }
        }
    }
}

func isDeviceActivelyUsed(_ device: AIBudsDevice) -> Bool {
    // Check if device is actively being used
    // Return true if actively used, false otherwise
    return false
}

Error Handling​

General Error Handling​

  • Handle errors gracefully
  • Provide clear error messages to users
  • Log errors for debugging

Objective-C​

// Handle SDK errors
- (void)handleSDKError:(NSError *)error {
    // Log the error
    [AIBudsLog error:@"SDK error: %@", error.localizedDescription];
    
    // Display error message to user
    UIAlertController *alert = [UIAlertController alertControllerWithTitle:@"Error" message:error.localizedDescription preferredStyle:UIAlertControllerStyleAlert];
    UIAlertAction *okAction = [UIAlertAction actionWithTitle:@"OK" style:UIAlertActionStyleDefault handler:nil];
    [alert addAction:okAction];
    [self presentViewController:alert animated:YES completion:nil];
    
    // Handle specific error types
    switch (error.code) {
        case AIBudsErrorCodeBluetoothDisabled:
            // Handle Bluetooth disabled error
            [self promptUserToEnableBluetooth];
            break;
        case AIBudsErrorCodeDeviceNotConnected:
            // Handle device not connected error
            [self promptUserToConnectDevice];
            break;
        case AIBudsErrorCodeNetworkError:
            // Handle network error
            [self promptUserToCheckNetwork];
            break;
        default:
            // Handle other errors
            break;
    }
}

Swift​

// Handle SDK errors
func handleSDKError(_ error: Error) {
    // Log the error
    AIBudsLog.error("SDK error: \(error.localizedDescription)")
    
    // Display error message to user
    let alert = UIAlertController(title: "Error", message: error.localizedDescription, preferredStyle: .alert)
    let okAction = UIAlertAction(title: "OK", style: .default, handler: nil)
    alert.addAction(okAction)
    present(alert, animated: true, completion: nil)
    
    // Handle specific error types
    if let error = error as? AIBudsError {
        switch error.code {
        case .bluetoothDisabled:
            // Handle Bluetooth disabled error
            promptUserToEnableBluetooth()
        case .deviceNotConnected:
            // Handle device not connected error
            promptUserToConnectDevice()
        case .networkError:
            // Handle network error
            promptUserToCheckNetwork()
        default:
            // Handle other errors
            break
        }
    }
}

Security Best Practices​

App Key Management​

  • Never hardcode app keys in your code
  • Use secure storage for app keys
  • Rotate app keys regularly

Objective-C​

// Get app key from secure storage
- (NSString *)getAppKey {
    // Use keychain to store app key
    NSDictionary *query = @{
        (__bridge id)kSecClass: (__bridge id)kSecClassGenericPassword,
        (__bridge id)kSecAttrAccount: @"AIBudsAppKey",
        (__bridge id)kSecReturnData: @YES
    };
    
    CFTypeRef result = NULL;
    OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query, &result);
    
    if (status == errSecSuccess) {
        NSData *data = (__bridge_transfer NSData *)result;
        return [[NSString alloc] initWithData:data encoding:NSUTF8StringEncoding];
    }
    
    return nil;
}

Swift​

// Get app key from secure storage
func getAppKey() -> String? {
    // Use keychain to store app key
    let query: [String: Any] = [
        kSecClass as String: kSecClassGenericPassword,
        kSecAttrAccount as String: "AIBudsAppKey",
        kSecReturnData as String: true
    ]
    
    var result: AnyObject?
    let status = SecItemCopyMatching(query as CFDictionary, &result)
    
    if status == errSecSuccess, let data = result as? Data {
        return String(data: data, encoding: .utf8)
    }
    
    return nil
}
