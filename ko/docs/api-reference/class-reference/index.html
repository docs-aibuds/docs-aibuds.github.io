Class Reference

This guide provides a reference for the main classes in the AIBuds SDK.

Core Classes​

AIBudsSDK​

The main entry point for the AIBuds SDK.

Objective-C​

@interface AIBudsSDK : NSObject

// Initialize the SDK
+ (void)initializeWithAppKey:(NSString *)appKey completion:(void (^)(BOOL success, NSError *error))completion;

// Get the shared instance
+ (instancetype)shared;

// Check if the SDK is initialized
@property (nonatomic, readonly, getter=isInitialized) BOOL initialized;

// Get the SDK version
@property (nonatomic, readonly) NSString *version;

// Set log level
+ (void)setLogLevel:(AIBudsLogLevel)level;

@end

Swift​

class AIBudsSDK: NSObject {
    
    // Initialize the SDK
    class func initialize(withAppKey appKey: String, completion: @escaping (Bool, Error?) -> Void)
    
    // Get the shared instance
    class func shared() -> AIBudsSDK
    
    // Check if the SDK is initialized
    var isInitialized: Bool { get }
    
    // Get the SDK version
    var version: String { get }
    
    // Set log level
    class func setLogLevel(_ level: AIBudsLogLevel)
}

AIBudsDevice​

Represents an AIBuds device.

Objective-C​

@interface AIBudsDevice : NSObject

// Device properties
@property (nonatomic, readonly) NSString *identifier;
@property (nonatomic, readonly) NSString *name;

@property (nonatomic, readonly) NSString *serialNumber;

@property (nonatomic, readonly) AIBudsDeviceConnectionState connectionState;
@property (nonatomic, readonly) NSInteger batteryLevel;
@property (nonatomic, readonly) NSInteger leftBatteryLevel;
@property (nonatomic, readonly) NSInteger rightBatteryLevel;
@property (nonatomic, readonly) NSInteger caseBatteryLevel;
@property (nonatomic, readonly, getter=isCharging) BOOL charging;

// Connect to the device
- (void)connectWithCompletion:(void (^)(BOOL success, NSError *error))completion;

// Disconnect from the device
- (void)disconnectWithCompletion:(void (^)(BOOL success, NSError *error))completion;

// Update device name
- (void)updateName:(NSString *)name completion:(void (^)(BOOL success, NSError *error))completion;

@end

class AIBudsDevice: NSObject {
